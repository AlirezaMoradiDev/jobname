def my_pow(x, n):
    if n < 0:
        return 1/recursive_pow(x, -1 * n)
    else:
        return recursive_pow(x, n)

def recursive_pow(x, n):
    if n == 0:
        return 1
    if n == 1:
        return x
    else:
        return x * recursive_pow(x, n-1)
